1. 변의 길이를 입력하면 해당하는 속이 빈 정사각형을 *로 그리는 프로그램을 작성하시오.

#include <stdio.h>
int main(void)
{
	int x,i,j;

	printf("변의 길이를 입력하시오:");
	scanf("%d", &x);

	for(i=0; i<x; i++)
		printf("*");
	printf("\n");
	for(j=0;j<x-2;j++)
	{
		printf("*");
		for(i=0;i<x-2;i++)
			printf(" ");
		printf("*\n");
	}
	for(i=0;i<x;i++)
		printf("*");
	printf("\n");

	getchar();
	getchar();
	getchar();

	return 0;
}

2. 정수 n(단 n은 홀수)을 입력하면 * 기호를 사용해서 해당하는 대각선의 길이가 n인 속이 꽉

   찬 마름모와 속이 빈 마름모를 옆으로 나란하게 그리는 프로그램을 작성하시오.
-----------------------------------------------------------------------------------------

#include <stdio.h>
int main(void)
{
	int n,i,e=1,j=1;

	printf("원하는 정수를 입력하시오:");
	scanf("%d",&n);

	while(j>0)
	{
		for(i=1;i<=n/2+1-j;i++)
			printf(" ");
		for(i=1;i<=j*2-1;i++)
			printf("*");
		for(i=1;i<=n/2+1-j;i++)
			printf("  ");
		printf("*");
		for(i=1;i<=j*2-3;i++)
			printf(" ");
		if(j==1)
			printf("\n");
		else
			printf("*\n");
		
		if(j==n/2)
			e=-1;
		j+=e;
	}
	getchar();
	getchar();
	getchar();
	return 0;
}

{

   printf("원하는 첫번째 좌표를 입력하시오:");

   scanf("%f,%f",&x2,&y2);

   d1=sqrt(x1*x1+y1*y1);
   d2=sqrt(x2*x2+y2*y2);

   if(d<d1)
   {
	   if(d<d2)
	   {
		   if(d1<d2)
			   printf("(%g,%g), (%g,%g), (%g,%g)" ,x,y,x1,y1,x2,y2);
		   else
			   printf("(%g,%g), (%g,%g), (%g,%g)" ,x,y,x2,y2,x1,y1);}
	   else
		   printf("(%g,%g), (%g,%g), (%g,%g)" ,x2,y2,x,y,x1,y1);}
   else
   {
	   if(d1<d2)

   printf("이다.");

   getchar();
   getchar();

입력된 세 정수들을 절대값이 가장 작은 수부터 순서대로 출력하는 프로그램을 
작성하시오. 단 관계연산자는 8번 이하로 사용하시오.  	(Ex. -1 -3 -2 : -1 -2 -3)
--------------------------------------------------------------------------------------

#include<stdio.h>
int main(void)
{
	int x,y,z,max,min;

	printf("원하는 숫자 3개를 입력하시오:");
	scanf("%d %d %d",&x,&y,&z);

	if( ((x>0)?x:-x) < ((y>0)?y:-y) )
	{
		if( ((y>0)?y:-y) < ((z>0)?z:-z) )
		{max=z; min=x;}
		else
		{
			if( ((x>0)?x:-x) < ((z>0)?z:-z) )
			{max=y; min=x;}
			else
			{max=y; min=z;}
		}
	}
	else
	{
		if( ((x>0)?x:-x) < ((z>0)?z:-z) )
		{max=z; min=y;}
		else
		{
			if( ((y>0)?y:-y) < ((z>0)?z:-z) )
			{max=x; min=y;}
			else
			{max=x; min=z;}
		}
	}
	printf("절대값이 가장 작은 수는 %d이고 가장 큰수는 %d이다.",min,max);

	getchar();
	getchar();
	getchar();
	return 0;
}
